#!/bin/bash
#script to report the usage of aws
#s3
#ec2
#lambda
#iam
#to list the s3 buckets
aws s3 sl > resourceTracker
# to list the ec2
aws ec2-instance describe-instances
aws ec2 describe-instances | jq '.Reservations[].Instances[].Instancesid'
# to list lambda
aws lambda list-functions > resourceTracker
#to list iam users
aws iam list-users > resourceTracker
