#!/bin/bash
#script to report the usage of aws-resources
#s3
#ec2
#lambda
#iam
#to list the s3 buckets
aws s3 sl > resourceTracker
# to list the ec2
aws ec2-instance describe-instances
 aws ec2 describe-instances --output json | jq -r '.Reservations[].Instances[].InstanceId'
# to list lambda
aws lambda list-functions > resourceTracker
#to list iam users
aws iam list-users > resourceTracker
